utils.FileUtils

描述：用于实现文件操作，读取，写入，读取Assets等

* 方法

boolean mkdir(String path);
建立目录，可以建立多层目录

void createFile(String path, String text);
创建一个空的文件

void rewriteFile(File file, String text);
void rewriteFile(String path, String text);
重写文件，这将覆盖掉原有的文件

void appendFile(File file, String text);
void appendFile(String path, String text);
追加文件，被追加的文件必须已存在

boolean deleteFile(String path);
删除一个文件

boolean deleteDir(String path);
删除一个目录

void copyFile(String source, String dest);
复制一个文件

void copyFolder(String source, String dest);
复制一个目录

void moveFile(String source, String dest);
移动一个文件

void moveFolder(String source, String dest);
移动一个目录

List<String> readFile(File file);
List<String> readFile(String path);
读取一个文件，按行生成List

String toString(List<String> list);
将读取的文件时得到的List转换成String

String readAssetFile(Context context, String fileName);
从assets内读取文件

String readFile(Context context, String path);
从程序私有目录读取文件

List<?> loadListFromFile(String path)
从文件中读取一个对象列表，该文件是随机对象文件

void saveListToFile(List<?> list, String path)
将一个对象列表写入文件，该文件是随机对象文件
